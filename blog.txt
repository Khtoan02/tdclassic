    <script>
        lucide.createIcons();

        // FILTER LOGIC
        const filterBtns = document.querySelectorAll('.filter-btn');
        const articles = document.querySelectorAll('.article-card');
        
        filterBtns.forEach(btn => {
            btn.addEventListener('click', () => {
                // Update Active State
                filterBtns.forEach(b => {
                    b.classList.remove('text-gold', 'border-gold');
                    b.classList.add('text-gray-500', 'border-transparent', 'hover:border-white/20');
                });
                btn.classList.remove('text-gray-500', 'border-transparent', 'hover:border-white/20');
                btn.classList.add('text-gold', 'border-gold');

                const filterValue = btn.getAttribute('data-filter');

                // Filter Items
                articles.forEach(article => {
                    if (filterValue === 'all' || article.getAttribute('data-category') === filterValue) {
                        article.style.display = 'flex'; // Show logic needs refinement for infinite scroll context, but keeping simple for demo
                    } else {
                        article.style.display = 'none';
                    }
                });
            });
        });

        // INFINITE SCROLL SIMULATION
        const hiddenItems = document.querySelectorAll('.hidden-item');
        const sentinel = document.getElementById('loading-sentinel');
        const endMessage = document.getElementById('end-of-content');
        let itemsLoaded = 0;
        const batchSize = 3; // Load 3 items at a time

        const observer = new IntersectionObserver((entries) => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    loadMoreArticles();
                }
            });
        }, { rootMargin: '100px' });

        observer.observe(sentinel);

        function loadMoreArticles() {
            // Show loading spinner
            sentinel.style.opacity = '1';

            // Simulate network delay
            setTimeout(() => {
                let loadedCount = 0;
                // Find currently hidden items and show them
                const currentlyHidden = Array.from(document.querySelectorAll('.article-card.hidden-item'));
                
                if (currentlyHidden.length === 0) {
                    sentinel.style.display = 'none';
                    endMessage.classList.remove('hidden');
                    return;
                }

                for (let i = 0; i < batchSize && i < currentlyHidden.length; i++) {
                    currentlyHidden[i].classList.remove('hidden-item');
                    // Add fade-in animation class if needed here
                    loadedCount++;
                }

                // Hide loader again
                sentinel.style.opacity = '0';

                // Check if we are done
                if (document.querySelectorAll('.article-card.hidden-item').length === 0) {
                    observer.unobserve(sentinel);
                    sentinel.style.display = 'none';
                    endMessage.classList.remove('hidden');
                }
            }, 1000); // 1 second delay
        }
    </script>
